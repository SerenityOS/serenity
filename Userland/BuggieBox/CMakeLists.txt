serenity_component(
    BuggieBox
    REQUIRED
    TARGETS BuggieBox
)

function (buggiebox_utility src)
    get_filename_component(utility ${src} NAME_WE)
    target_sources(BuggieBox PRIVATE ${src})
    set_source_files_properties(${src} PROPERTIES COMPILE_DEFINITIONS "serenity_main=${utility}_main")
endfunction()
